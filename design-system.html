Usage Examples

Basic Component

import { responsiveSpacing, typography, buttonSizes } from "@/constants"; export function MySection() { return ( <section className={responsiveSpacing.sectionY}> <div className={`container mx-auto ${responsiveSpacing.containerX}`}> <h2 className={typography.heading.h1}> Section Title </h2> <Button className={buttonSizes.sizes.lg}> Action </Button> </div> </section> ); }

Using Component Presets

import { components } from "@/constants"; <div className={components.sectionHeader.container}> <h2 className={components.sectionHeader.title}> Title </h2> <p className={components.sectionHeader.description}> Description </p> </div>

Complete Example

import { responsiveSpacing, typography, buttonSizes, gaps, components, containers } from "@/constants"; export function Hero() { return ( <section className={`relative min-h-screen flex items-center ${responsiveSpacing.containerX}`}> <div className={`container mx-auto ${responsiveSpacing.blockGap}`}> <div className={containers.xl}> <h1 className={`${typography.display["2xl"]} ${typography.weight.bold}`}> Hero Title </h1> <p className={`${typography.body.xl} ${containers.lg} mx-auto`}> Hero description </p> <div className={`flex gap ${gaps.responsive.lg}`}> <Button className={buttonSizes.sizes.xl}> Get Started </Button> </div> </div> </div> </section> ); }
